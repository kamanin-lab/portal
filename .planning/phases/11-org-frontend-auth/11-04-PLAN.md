    approveTask: vi.fn(),
    requestChanges: vi.fn(),
    putOnHold: vi.fn(),
    resumeTask: vi.fn(),
    cancelTask: vi.fn(),
    isLoading: false,
  }),
}))

import { TaskActions } from '../components/TaskActions'

const wrapper = ({ children }: { children: ReactNode }) =>
  createElement(
    QueryClientProvider,
    { client: new QueryClient({ defaultOptions: { queries: { retry: false } } }) },
    children
  )

describe('TaskActions viewer guard', () => {
  beforeEach(() => {
    vi.clearAllMocks()
  })

  test('renders nothing when isViewer is true', () => {
    orgMock.useOrg.mockReturnValue({ organization: null, orgRole: 'viewer', isAdmin: false, isMember: false, isViewer: true, isLoading: false })
    const { container } = render(
      createElement(TaskActions, { taskId: 'task-1', status: 'to do' }),
      { wrapper }
    )
    expect(container).toBeEmptyDOMElement()
  })

  test('renders action buttons when isViewer is false', () => {
    orgMock.useOrg.mockReturnValue({ organization: null, orgRole: 'member', isAdmin: false, isMember: true, isViewer: false, isLoading: false })
    const { container } = render(
      createElement(TaskActions, { taskId: 'task-1', status: 'client review' }),
      { wrapper }
    )
    expect(container).not.toBeEmptyDOMElement()
  })
})
