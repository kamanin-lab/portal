    **STEP 1:** Create `scripts/check-german-only.mjs`:

    ```javascript
    #!/usr/bin/env node
    // German-only content gate for built widget artifacts (WIDG-QA-04).
    // Scans rendered HTML body text (strips <script>/<style>/tags) for English-word blacklist.
    // Exit 1 on any hit so CI blocks the merge.

    import { readFileSync, readdirSync, statSync } from "node:fs";
    import { join } from "node:path";

    const DIST_DIR = "dist/widgets";

    // Whole-word matches only (word boundaries). Case-insensitive.
    const BLACKLIST = [
      "Loading",
      "Retry",
      "Reload",
      "Submit",
      "Cancel",
      "Close",
      "Details",
      // 'Error' is a JS built-in class name — only flag when it appears as user-facing text.
      // We strip <script> before scanning, but minified HTML may still contain Error identifiers
      // in inline event handlers. We match on trailing punctuation ('Error.', 'Error ', 'Error!')
      // to reduce false positives.
    ];

    // Extract visible text from an HTML file: drop <script>, <style>, tags; keep inner text.
    function extractText(html) {
      const noScript = html.replace(/<script[\s\S]*?<\/script>/gi, " ");
      const noStyle = noScript.replace(/<style[\s\S]*?<\/style>/gi, " ");
      const noTags = noStyle.replace(/<[^>]+>/g, " ");
      // Collapse whitespace
      return noTags.replace(/\s+/g, " ").trim();
    }

    function findWidgetHtmlFiles(dir) {
      const results = [];
      if (!statSync(dir, { throwIfNoEntry: false })) return results;
      for (const entry of readdirSync(dir)) {
        const path = join(dir, entry);
        const s = statSync(path);
        if (s.isDirectory()) {
          results.push(...findWidgetHtmlFiles(path));
        } else if (entry.endsWith(".html")) {
          results.push(path);
        }
      }
      return results;
    }

    function scanFile(path) {
      const html = readFileSync(path, "utf8");
      const text = extractText(html);
      const hits = [];
      for (const word of BLACKLIST) {
        const pattern = new RegExp(`\\b${word}\\b`, "gi");
        const matches = text.match(pattern);
        if (matches) {
          hits.push({ word, count: matches.length, path });
        }
      }
      return hits;
    }

    const files = findWidgetHtmlFiles(DIST_DIR);
    if (files.length === 0) {
      console.error(`[check:german] No widget HTML files found under ${DIST_DIR}. Run npm run build:widgets first.`);
      process.exit(1);
    }

    let totalHits = 0;
    for (const file of files) {
      const hits = scanFile(file);
      for (const hit of hits) {
        console.error(`[check:german] FAIL — ${hit.path} contains blacklisted English word "${hit.word}" (${hit.count} match${hit.count === 1 ? "" : "es"})`);
        totalHits += hit.count;
      }
    }

    if (totalHits > 0) {
      console.error(`[check:german] Total blacklist hits: ${totalHits}. Widget must be German-only (WIDG-QA-04).`);
      process.exit(1);
    }

    console.log(`[check:german] OK — scanned ${files.length} file(s), no English blacklist matches.`);
    process.exit(0);
    ```
