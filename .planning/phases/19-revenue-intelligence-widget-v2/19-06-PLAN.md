    ```tsx
    import type { Block, RepeatData } from "../lib/types";
    import { useCountUp } from "../lib/useCountUp";
    import { BlockSkeleton } from "./BlockSkeleton";

    export function RepeatBlock({ block }: { block: Block<RepeatData> }) {
      if (block.status === "error") {
        return <BlockSkeleton variant="error" approxHeight={180} />;
      }
      const { data } = block;
      const rate = useCountUp(data.repeat_rate_pct);
      const trend = data.trend_pp;
      const showTrend = typeof trend === "number" && trend !== 0;

      return (
        <section
          className="p-6"
          style={{ background: "var(--color-surface)", borderRadius: "var(--radius-lg)", border: "1px solid var(--color-border)" }}
        >
          <h2 className="text-lg font-semibold leading-snug">Wiederkäufer — letzte 90 Tage</h2>

          <div className="flex items-baseline justify-between mt-4">
            <span className="text-[15px]">Wiederkaufrate</span>
            <div className="flex items-baseline gap-2">
              <span className="text-3xl font-semibold tabular-nums">{Math.round(rate)} %</span>
              {showTrend && (
                <span
                  className="text-[13px]"
                  style={{ color: trend > 0 ? "var(--color-success)" : "var(--color-danger)" }}
                >
                  {trend > 0 ? `↑ +${trend} PP` : `↓ ${trend} PP`}
                </span>
              )}
            </div>
          </div>

          <div className="text-[15px] mt-3">
            Branchen-Benchmark ~{Math.round(data.benchmark_pct)} %
          </div>
          <div className="text-[13px]" style={{ color: "var(--color-subtle)" }}>
            (Shopify B2C, 2024)
          </div>

          {typeof data.median_days_to_2nd === "number" && (
            <div className="flex items-baseline justify-between mt-3">
              <span className="text-[13px]" style={{ color: "var(--color-muted)" }}>
                Ø Tage bis 2. Kauf
              </span>
              <span className="text-[15px]">{data.median_days_to_2nd} Tage</span>
            </div>
          )}

          <div className="text-[13px] mt-3" style={{ color: "var(--color-subtle)" }}>
            Basis: {data.total_orders.toLocaleString("de-DE")} Bestellungen in 90 Tagen
          </div>
        </section>
      );
    }
    ```

    ```tsx
    import type { Block, BasketData, BasketPair, AovBand } from "../lib/types";
    import { formatCurrency } from "../lib/formatters";
    import { useCountUp } from "../lib/useCountUp";
    import { BlockSkeleton } from "./BlockSkeleton";

    export function BasketOrAovBlock({ block }: { block: Block<BasketData> }) {
      if (block.status === "error") {
        return <BlockSkeleton variant="error" approxHeight={240} />;
      }
      const { data } = block;
      if (data.mode === "aov_bands") {
        return <AovBandsView data={data} />;
      }
      const header =
        data.mode === "market_basket_category"
          ? "Häufig zusammen gekauft (Kategorien)"
          : "Häufig zusammen gekauft";
      return <BasketView data={data} header={header} />;
    }

    // ─── market_basket_product / market_basket_category mode ─────────────────

    function BasketView({ data, header }: { data: BasketData; header: string }) {
      const pairs = (data.basket_pairs ?? []).slice(0, 3);
      return (
        <section
          className="p-6"
          style={{ background: "var(--color-surface)", borderRadius: "var(--radius-lg)", border: "1px solid var(--color-border)" }}
        >
          <h2 className="text-lg font-semibold leading-snug">{header}</h2>
          <div className="flex flex-col gap-4 mt-4">
            {pairs.map((p, i) => (
              <BasketPairRow key={i} pair={p} />
            ))}
          </div>
          <div className="text-[13px] mt-4" style={{ color: "var(--color-subtle)" }}>
            Basis: letzte 90 Tage
          </div>
        </section>
      );
    }

    function BasketPairRow({ pair }: { pair: BasketPair }) {
      const confPct = Math.round(pair.confidence * 100);
      const supPct = Math.round(pair.support * 100);
      const liftStr = pair.lift.toFixed(1).replace(".", ",");
      return (
        <div className="flex flex-col gap-1">
          <div className="text-[15px]" style={{ color: "var(--color-fg)" }}>
            {pair.a_name} <span style={{ color: "var(--color-accent)" }}>+</span> {pair.b_name}
          </div>
          <div className="text-[15px]" style={{ color: "var(--color-muted)" }}>
            {`${confPct}% der ${pair.a_name}-Käufer kauften auch ${pair.b_name}`}
          </div>
          <div className="text-[13px] font-mono" style={{ color: "var(--color-subtle)" }}>
            {`Support ${supPct}% · Konfidenz ${confPct}% · Lift ${liftStr}×`}
          </div>
        </div>
      );
    }

    // ─── aov_bands mode ──────────────────────────────────────────────────────

    function AovBandsView({ data }: { data: BasketData }) {
      const avg = useCountUp(data.avg_order_value);
      return (
        <section
          className="p-6"
          style={{ background: "var(--color-surface)", borderRadius: "var(--radius-lg)", border: "1px solid var(--color-border)" }}
        >
          <h2 className="text-lg font-semibold leading-snug">Bestellwert-Verteilung — letzte 90 Tage</h2>
          <div className="flex flex-col gap-3 mt-4">
            {data.aov_bands.map((band, i) => (
              <AovBandRow key={i} band={band} />
            ))}
          </div>
          <hr className="my-4" style={{ borderTop: "1px solid var(--color-border)" }} />
          <div className="flex items-baseline justify-between">
            <span className="text-[13px]" style={{ color: "var(--color-muted)" }}>Ø Bestellwert</span>
            <span className="text-2xl font-semibold tabular-nums">{formatCurrency(avg)}</span>
          </div>
          {typeof data.median_order_value === "number" && (
            <div className="flex items-baseline justify-between mt-2">
              <span className="text-[13px]" style={{ color: "var(--color-muted)" }}>Median Bestellwert</span>
              <span className="text-[15px] tabular-nums">{formatCurrency(data.median_order_value)}</span>
            </div>
          )}
        </section>
      );
    }

    function AovBandRow({ band }: { band: AovBand }) {
      const countPct = Math.round(band.share_of_count * 100);
      const revPct = Math.round(band.share_of_revenue * 100);
      return (
        <div className="flex flex-col gap-1">
          <div className="flex justify-between text-[13px]">
            <span>{band.label}</span>
            <span className="tabular-nums">{`${countPct} % · ${revPct} % Umsatz`}</span>
          </div>
          <div
            style={{
              height: 6,
              width: `${countPct}%`,
              background: "color-mix(in oklch, var(--color-accent) 50%, transparent)",
              borderRadius: 3,
            }}
          />
          <div
            style={{
              height: 6,
              width: `${revPct}%`,
              background: "var(--color-accent)",
              borderRadius: 3,
            }}
          />
        </div>
      );
    }
    ```
