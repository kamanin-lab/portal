       ```tsx
       function computeQuintiles(nonZeroCounts: number[]): number[] {
         if (nonZeroCounts.length === 0) return [0, 0, 0, 0];
         const sorted = [...nonZeroCounts].sort((a, b) => a - b);
         const q = (p: number) => sorted[Math.floor((sorted.length - 1) * p)] ?? 0;
         return [q(0.2), q(0.4), q(0.6), q(0.8)];
       }
       function colorForCount(count: number, quintiles: number[], isBest: boolean): string {
         if (count === 0) return "color-mix(in oklch, var(--color-subtle) 12%, transparent)";
         const [q1, q2, q3, q4] = quintiles;
         let pct = 20;
         if (count > q4 || isBest) pct = 85;
         else if (count > q3) pct = 60;
         else if (count > q2) pct = 40;
         else if (count > q1) pct = 40;
         return `color-mix(in oklch, var(--color-accent) ${pct}%, transparent)`;
       }
       ```

    5. **Refetch function** (V4 gate — only `weekly_heatmap` tool name):
       ```tsx
       async function refetch(weeks: 4 | 8 | 12) {
         if (!app) return;
         setView((v) => ({ ...v, loading: true, error: null }));
         try {
           const result = await app.callServerTool({ name: "weekly_heatmap", arguments: { weeks } });
           const structured = (result as { structuredContent?: unknown }).structuredContent;
           if (isHeatmapData(structured)) {
             setView({ weeks, data: structured, error: null, loading: false });
           } else {
             setView((v) => ({ ...v, loading: false, error: "invalid payload" }));
           }
         } catch (err) {
           setView((v) => ({ ...v, loading: false, error: err instanceof Error ? err.message : String(err) }));
         }
       }
       ```
