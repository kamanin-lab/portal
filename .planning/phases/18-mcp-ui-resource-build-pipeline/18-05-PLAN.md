```typescript
// URL query-string fixture-mode parser for widget dev/test.
// Consumed by Phase 19 block components to force specific data shapes (basket-aov, one-block-failing).
// Phase 18 ships the parser alongside the harness for stability; v1 widget does not consume it.

export type FixtureMode = 'basket-aov' | 'one-block-failing'

export function getFixtureMode(): FixtureMode | null {
  const params = new URLSearchParams(window.location.search)
  const mode = params.get('mock')
  if (mode === 'basket-aov' || mode === 'one-block-failing') return mode
  return null
}
```
