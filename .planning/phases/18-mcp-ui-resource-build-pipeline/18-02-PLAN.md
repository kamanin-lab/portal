// widgets/shared/vite.base.ts
// KEEP plugin order stable: react() → tailwindcss() → viteSingleFile()
// Reordering breaks CSS inlining (Pitfall 3).
import type { UserConfig } from 'vite'
import react from '@vitejs/plugin-react'
import tailwindcss from '@tailwindcss/vite'
import { viteSingleFile } from 'vite-plugin-singlefile'
import { resolve } from 'node:path'
import { fileURLToPath } from 'node:url'
import { dirname } from 'node:path'

const here = dirname(fileURLToPath(import.meta.url))

export interface BuildWidgetOpts {
  root: string
  outFileName: string  // e.g. 'daily-briefing.html' — flat output per Pitfall 7
}

export function buildWidgetConfig(opts: BuildWidgetOpts): UserConfig {
  return {
    plugins: [
      react(),
      tailwindcss(),
      viteSingleFile({ removeViteModuleLoader: true }),
    ],
    root: opts.root,
    build: {
      outDir: resolve(here, '../../dist/widgets'),  // ABSOLUTE path from shared/ → mcp-poc/dist/widgets
      emptyOutDir: false,                            // build-widgets.mjs empties once at start
      assetsInlineLimit: 100_000_000,
      cssCodeSplit: false,
      rollupOptions: {
        output: {
          inlineDynamicImports: true,
          entryFileNames: opts.outFileName,          // FLAT: dist/widgets/daily-briefing.html (not dist/widgets/daily-briefing/index.html)
        },
      },
    },
  }
}

// ============================================================
// PREACT FALLBACK — DOCUMENTATION-ONLY per MCPAPP-BUILD-05
// ============================================================
// Activate when the 300 KB gz bundle budget busts (measured by
// scripts/check-widget-bundle-size.mjs):

//        'react-dom/test-utils':  'preact/test-utils',

Bundle size check script shape (from 18-RESEARCH.md §Bundle Budget Measurement):
```javascript
// scripts/check-widget-bundle-size.mjs
import { readdirSync, readFileSync, existsSync } from 'node:fs'
import { gzipSync } from 'node:zlib'
import { resolve, join } from 'node:path'

const BUDGET_BYTES = 307200  // 300 * 1024

const distDir = resolve(process.cwd(), 'dist/widgets')
if (!existsSync(distDir)) {
  console.error('✗ dist/widgets not found — run npm run build:widgets first')
  process.exit(1)
}

const htmlFiles = readdirSync(distDir).filter((f) => f.endsWith('.html'))
if (htmlFiles.length === 0) {
  console.error('✗ No widget HTML files found in dist/widgets')
  process.exit(1)
}

let failed = false
for (const file of htmlFiles) {
  const buf = readFileSync(join(distDir, file))
  const gzSize = gzipSync(buf).length
  const kb = (gzSize / 1024).toFixed(1)
  const pass = gzSize <= BUDGET_BYTES
  const mark = pass ? '✓' : '✗'
  console.log(`${mark} ${file}: ${gzSize} bytes (${kb} KB gz) — budget 307200 bytes (300 KB gz)`)
  if (!pass) failed = true
}

if (failed) {
  console.error('\n✗ One or more widgets exceed the 300 KB gz budget (MCPAPP-BUILD-04)')
  console.error('  See widgets/shared/vite.base.ts Preact fallback documentation.')
  process.exit(1)
}

console.log('\n✓ All widgets within 300 KB gz budget.')
```
