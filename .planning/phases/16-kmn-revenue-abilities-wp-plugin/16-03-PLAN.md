**execute_callback structure:**

```php
add_action( 'wp_abilities_api_init', function () {
    if ( ! function_exists( 'wp_register_ability' ) ) return;

    wp_register_ability( 'kmn/weekly-briefing-data', [
        'label'       => 'Weekly Briefing Data',
        'description' => 'Combined payload for the Monday briefing email. Aggregates last-week summary, best slot, repeat metrics, and top 3 products.',
        'category'    => 'woocommerce',
        'meta'        => [ 'show_in_rest' => true, 'mcp' => [ 'public' => true ] ],
        'input_schema' => [
            'type' => 'object',
            'properties' => [
                'reference_date' => [ 'type' => 'string', 'pattern' => '^\\d{4}-\\d{2}-\\d{2}$' ],
                'timezone'       => [ 'type' => 'string' ],
            ],
            'required' => [],
        ],
        'output_schema' => [ /* per shape above */ ],

        'execute_callback' => function ( $input ) {
            try {
                $key = kmn_revenue_cache_key( 'kmn/weekly-briefing-data', $input );
                return kmn_revenue_cached( $key, KMN_REVENUE_TTL_BRIEFING, function () use ( $input ) {
                    global $wpdb;
                    kmn_revenue_set_query_timeout_ms( 2000 );

                    $offset    = kmn_revenue_resolve_tz_offset( $input );
                    $ref_date  = sanitize_text_field( $input['reference_date'] ?? current_time( 'Y-m-d' ) );
                    $statuses  = [ 'wc-completed', 'wc-processing' ];
                    $placeholders = kmn_revenue_prepare_in_placeholders( $statuses );

                    // 1. Sub-ability: best slot from last week
                    $heatmap_ability = wp_get_ability( 'kmn/weekly-heatmap' );
                    if ( ! $heatmap_ability ) {
                        return kmn_revenue_response( false, [], 'sub-ability kmn/weekly-heatmap not registered' );
                    }
                    $heatmap_resp = call_user_func(
                        $heatmap_ability->get_callback( 'execute' ),
                        [ 'weeks' => 8, 'reference_date' => $ref_date, 'timezone' => $input['timezone'] ?? null ]
                    );
                    if ( empty( $heatmap_resp['success'] ) ) {
                        return kmn_revenue_response( false, [], 'weekly-briefing-data sub-call failed: ' . ( $heatmap_resp['error'] ?? 'unknown' ) );
                    }
                    $best_slot = $heatmap_resp['data']['best_slot'] ?? null;

                    // 2. Sub-ability: repeat metrics
                    $repeat_ability = wp_get_ability( 'kmn/repeat-metrics' );
                    if ( ! $repeat_ability ) {
                        return kmn_revenue_response( false, [], 'sub-ability kmn/repeat-metrics not registered' );
                    }
                    $repeat_resp = call_user_func(
                        $repeat_ability->get_callback( 'execute' ),
                        [ 'days' => 90, 'reference_date' => $ref_date, 'timezone' => $input['timezone'] ?? null ]
                    );
                    if ( empty( $repeat_resp['success'] ) ) {
                        return kmn_revenue_response( false, [], 'weekly-briefing-data sub-call failed: ' . ( $repeat_resp['error'] ?? 'unknown' ) );
                    }
                    $repeat_data = [
                        'repeat_rate_pct'     => $repeat_resp['data']['repeat_rate_pct'],
                        'total_orders'        => $repeat_resp['data']['total_orders'],
                        'unique_customers'    => $repeat_resp['data']['unique_customers'],
                        'returning_customers' => $repeat_resp['data']['returning_customers'],
                        'median_days_to_2nd'  => $repeat_resp['data']['median_days_to_2nd'],
                    ];

                    // 3. Custom: last-week summary (Mon-Sun rolling, anchored on $ref_date - 7d)
                    $ref_dt = new DateTimeImmutable( $ref_date, new DateTimeZone( $offset ) );
                    // Find Monday of last week:
                    $iso_dow = (int) $ref_dt->format( 'N' );  // 1..7 (Mon..Sun)
                    $this_monday = $ref_dt->modify( '-' . ( $iso_dow - 1 ) . ' days' );
                    $last_monday = $this_monday->modify( '-7 days' );
                    $last_sunday_end = $this_monday;  // exclusive upper bound
                    $prior_monday = $last_monday->modify( '-7 days' );

                    $week_sql = $wpdb->prepare(
                        "SELECT COUNT(*) AS cnt, COALESCE(SUM(s.net_total), 0) AS total
                         FROM   {$wpdb->prefix}wc_order_stats s
                         WHERE  s.date_created >= %s AND s.date_created < %s
                           AND  s.status IN ($placeholders)",
                        array_merge(
                            [ $last_monday->format( 'Y-m-d H:i:s' ), $last_sunday_end->format( 'Y-m-d H:i:s' ) ],
                            $statuses
                        )
                    );
                    $cur = $wpdb->get_row( $week_sql, ARRAY_A );

                    $prior_sql = $wpdb->prepare(
                        "SELECT COUNT(*) AS cnt, COALESCE(SUM(s.net_total), 0) AS total
                         FROM   {$wpdb->prefix}wc_order_stats s
                         WHERE  s.date_created >= %s AND s.date_created < %s
                           AND  s.status IN ($placeholders)",
                        array_merge(
                            [ $prior_monday->format( 'Y-m-d H:i:s' ), $last_monday->format( 'Y-m-d H:i:s' ) ],
                            $statuses
                        )
                    );
                    $prior = $wpdb->get_row( $prior_sql, ARRAY_A );

                    $cur_total   = (float) $cur['total'];
                    $cur_count   = (int)   $cur['cnt'];
                    $prior_total = (float) $prior['total'];
                    $aov         = $cur_count > 0 ? round( $cur_total / $cur_count, 2 ) : 0.0;
                    $vs_prior    = $prior_total > 0 ? round( ( $cur_total / $prior_total - 1 ) * 100, 1 ) : 0.0;

                    $last_week_summary = [
                        'week_start'        => $last_monday->format( 'Y-m-d' ),
                        'week_end'          => $last_sunday_end->modify( '-1 second' )->format( 'Y-m-d' ),
                        'revenue_total'     => $cur_total,
                        'order_count'       => $cur_count,
                        'aov'               => $aov,
                        'vs_prior_week_pct' => $vs_prior,
                    ];

                    // 4. Custom: top 3 products by quantity sold last week
                    $top_sql = $wpdb->prepare(
                        "SELECT p.product_id, SUM(p.product_qty) AS qty_sold
                         FROM   {$wpdb->prefix}wc_order_product_lookup p
                         JOIN   {$wpdb->prefix}wc_order_stats s ON p.order_id = s.order_id
                         WHERE  s.date_created >= %s AND s.date_created < %s
                           AND  s.status IN ($placeholders)
                         GROUP  BY p.product_id
                         ORDER  BY qty_sold DESC
                         LIMIT  3",
                        array_merge(
                            [ $last_monday->format( 'Y-m-d H:i:s' ), $last_sunday_end->format( 'Y-m-d H:i:s' ) ],
                            $statuses
                        )
                    );
                    $top_rows = $wpdb->get_results( $top_sql, ARRAY_A );
                    // Batch-load names
                    $pids = array_map( fn( $r ) => (int) $r['product_id'], $top_rows );
                    $names = [];
                    if ( $pids ) {
                        $posts = get_posts( [ 'post_type' => 'product', 'post__in' => $pids, 'posts_per_page' => -1, 'post_status' => 'any', 'orderby' => 'post__in' ] );
                        foreach ( $posts as $p ) $names[ $p->ID ] = $p->post_title;
                    }
                    $top_products_3 = array_map( function ( $r ) use ( $names ) {
                        $pid = (int) $r['product_id'];
                        return [
                            'product_id' => $pid,
                            'name'       => $names[ $pid ] ?? '(unknown)',
                            'qty_sold'   => (int) $r['qty_sold'],
                        ];
                    }, $top_rows );

                    return kmn_revenue_response( true, [
                        'last_week_summary' => $last_week_summary,
                        'best_slot'         => $best_slot,
                        'repeat_metrics'    => $repeat_data,
                        'top_products_3'    => $top_products_3,
                        'calculated_at'     => gmdate( 'c' ),
                    ] );
                }, ! empty( $input['_skip_cache'] ) );
            } catch ( \Throwable $e ) {
                return kmn_revenue_response( false, [], 'weekly-briefing-data failed: ' . $e->getMessage() );
            }
        },
        'permission_callback' => function () { return current_user_can( 'manage_woocommerce' ); },
    ] );
} );
```

### 2.1 `scripts/verify-wp-bridge.sh` (~150 lines)

Bash script with:

```bash
#!/bin/bash
set -euo pipefail

: "${BRIDGE_URL:=https://summerfield.ddev.site/wp-json/mcp/kmn-revenue}"
: "${WP_USER:=dev-admin}"
: "${WP_APP_PASS:?WP_APP_PASS env var required}"
: "${SUBSCRIBER_USER:=}"        # optional: a customer/subscriber user for the permission test
: "${SUBSCRIBER_APP_PASS:=}"    # optional

AUTH="${WP_USER}:${WP_APP_PASS}"
CURL=(-sk -H 'Content-Type: application/json' -H 'Accept: application/json,text/event-stream')
RED='\033[0;31m'; GREEN='\033[0;32m'; NC='\033[0m'

fail() { echo -e "${RED}FAIL${NC}: $*" >&2; exit 1; }
ok()   { echo -e "${GREEN}OK${NC}: $*"; }

# 1. tools/list returns exactly 5 sanitized tool names
list=$(curl "${CURL[@]}" -u "$AUTH" -X POST "$BRIDGE_URL" \
  -d '{"jsonrpc":"2.0","id":1,"method":"tools/list"}')
count=$(echo "$list" | jq '.result.tools | length')
[[ "$count" == "5" ]] || fail "expected 5 tools, got $count"

names=$(echo "$list" | jq -r '.result.tools[].name' | sort | tr '\n' ',')
expected="kmn-market-basket,kmn-repeat-metrics,kmn-revenue-run-rate,kmn-weekly-briefing-data,kmn-weekly-heatmap,"
[[ "$names" == "$expected" ]] || fail "tool name mismatch:\n  got:      $names\n  expected: $expected"
ok "tools/list returns 5 sanitized names (slash→hyphen via McpNameSanitizer)"

# 2. Per-tool shape + value assertions
call() {
  local tool="$1" args="$2"
  curl "${CURL[@]}" -u "$AUTH" -X POST "$BRIDGE_URL" \
    -d "{\"jsonrpc\":\"2.0\",\"id\":42,\"method\":\"tools/call\",\"params\":{\"name\":\"$tool\",\"arguments\":$args}}"
}
unwrap() { echo "$1" | jq -r '.result.content[0].text'; }

# 2a. weekly-heatmap: best_slot Do 20:00 / order_count ∈ [17,21]

data=$(unwrap "$resp" | jq '.data')
```

```bash
#!/bin/bash
# Static lint for SQL safety + HPOS-correctness inside abilities/.
# Covers ABIL-DEF-06 (no raw $input interpolation) and ABIL-QA-04 (HPOS-safe).

# 4. ABIL-QA-03 (timeout): every ability calls kmn_revenue_set_query_timeout_ms
```
