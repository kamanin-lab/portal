Slug derivation (exact SQL logic from Phase 9 migration, translated to TypeScript):
```typescript
// SQL: lower(regexp_replace(split_part(email, '@', 2), '\.[^.]+$', ''))
function deriveOrgSlug(email: string): string {
  const domain = email.split('@')[1] ?? email;
  return domain.replace(/\.[^.]+$/, '').toLowerCase().slice(0, 30);
}

// Uniqueness: query organizations for slug, append -2/-3 up to 5 attempts
async function deriveUniqueSlug(supabase, email: string, override?: string): Promise<string> {
  const base = override ?? deriveOrgSlug(email);
  for (let i = 1; i <= 5; i++) {
    const candidate = i === 1 ? base : `${base}-${i}`;
    const { data } = await supabase.from('organizations').select('id').eq('slug', candidate).maybeSingle();
    if (!data) return candidate;
  }
  throw new Error(`Cannot derive unique slug for "${base}" — 5 variants already taken`);
}
```
