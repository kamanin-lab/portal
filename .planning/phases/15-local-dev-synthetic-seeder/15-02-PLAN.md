**Class structure:**
```php
final class KMN_Seeder_Command {

    // ---- Constants (copy all distribution arrays from <interfaces> block) ----
    const HOUR_WEIGHTS   = [ /* ... */ ];
    const DOW_SHARES     = [ /* ... */ ];
    const BASKET_SIZE_SHARES = [ /* ... */ ];
    const AOV_BANDS      = [ /* ... */ ];
    const PAYMENT_METHODS = [ /* ... */ ];
    const STATUS_SHARES  = [ /* ... */ ];
    const COHORT         = [ /* ... */ ];
    const BASKET_PAIRS   = [ /* ... */ ];

    const META_TEST_ORDER = '_kmn_test_order';
    const META_TEST_USER  = '_kmn_test_user';

    /**
     * wp kmn seed [<subcommand>] [--flags]
     *
     * Default subcommand: generate orders.
     */
    public function seed( $args, $assoc_args ) {
        $this->guard_environment();   // SEED-07 — hard fail outside *.ddev.site

        // Subcommand dispatch: `wp kmn seed reset`
        if ( isset( $args[0] ) && $args[0] === 'reset' ) {
            $this->reset( $assoc_args );
            return;
        }

        // Parse flags with defaults
        $weeks          = max( 4, min( 26, (int)( $assoc_args['weeks']       ?? 12 ) ) );
        $daily_avg      = max( 1, (int)( $assoc_args['daily-avg']            ?? 15 ) );
        $daily_stddev   = max( 0, (int)( $assoc_args['daily-stddev']         ?? 5  ) );
        $repeat_rate    = max( 0.0, min( 1.0, (float)( $assoc_args['repeat-rate']     ?? 0.22 ) ) );
        $multi_item     = max( 0.0, min( 1.0, (float)( $assoc_args['multi-item-rate'] ?? 0.25 ) ) );
        $dry_run        = isset( $assoc_args['dry-run'] );
        $rng_seed       = isset( $assoc_args['seed'] ) ? (int) $assoc_args['seed'] : null;

        if ( $rng_seed !== null ) {
            mt_srand( $rng_seed );
        }

        // Performance hygiene — SEEDER_SPEC §3e
        update_option( 'woocommerce_email_new_order_enabled', 'no' );
        if ( ! defined( 'WP_IMPORTING' ) ) {
            define( 'WP_IMPORTING', true );
        }

        // Plan daily counts via Poisson-like distribution
        $daily_plan = $this->plan_daily_counts( $weeks, $daily_avg, $daily_stddev );  // returns [date => count]
        $total_orders = array_sum( $daily_plan );

        WP_CLI::log( sprintf(
            "Plan: %d weeks, avg %d/day, target ~%d orders (repeat=%0.2f, multi-item=%0.2f)%s",
            $weeks, $daily_avg, $total_orders, $repeat_rate, $multi_item,
            $dry_run ? ' [DRY RUN]' : ''
        ) );

        if ( $dry_run ) {
            $this->print_plan_summary( $daily_plan, $repeat_rate, $multi_item );
            return;
        }

        // Generate customers up-front (~900) — 78% one-time, 22% repeaters split per COHORT
        $customers = $this->generate_customers( $total_orders, $repeat_rate );
        WP_CLI::log( sprintf( 'Generated %d synthetic customer records.', count( $customers ) ) );

        // Load product pool with name-based basket-pair mapping
        $products = $this->load_product_pool();
        if ( count( $products ) < 10 ) {
            WP_CLI::error( sprintf(
                'Product catalog too small (%d products). Import Summerfield products first.',
                count( $products )
            ) );
        }

        // Emit orders day-by-day with progress bar
        $progress = \WP_CLI\Utils\make_progress_bar( 'Seeding orders', $total_orders );
        $created  = 0;

        foreach ( $daily_plan as $date_str => $count_for_day ) {
            $dow = wp_date( 'D', strtotime( $date_str ) );
            $hour_weights = $this->hour_weights_for_dow( $dow );
            for ( $i = 0; $i < $count_for_day; $i++ ) {
                $hour   = $this->weighted_pick( $hour_weights );
                $minute = mt_rand( 0, 59 );
                $dt     = sprintf( '%s %02d:%02d:%02d', $date_str, $hour, $minute, mt_rand( 0, 59 ) );

                $customer = $this->pick_customer_for_order( $customers, $date_str );
                $items    = $this->compose_basket( $products, $multi_item );
                $status   = $this->weighted_pick( self::STATUS_SHARES );
                $payment  = $this->pick_payment_method();

                $this->create_order( $dt, $customer, $items, $status, $payment );
                $created++;
                $progress->tick();
            }
        }
        $progress->finish();

        // Force-sync HPOS stats table per SEEDER_SPEC §3b
        WP_CLI::runcommand( 'wc tool run regenerate_order_stats --user=admin', [ 'launch' => false, 'exit_error' => false ] );

        WP_CLI::success( sprintf( 'Seeded %d orders across %d days.', $created, count( $daily_plan ) ) );
    }

    // ---- reset, guard_environment, helpers, etc. ----
}

WP_CLI::add_command( 'kmn', 'KMN_Seeder_Command' );
```

4. **`compose_basket($products, $multi_item_rate)` (SEED-04)** — enforces pre-seeded basket pairs:
```php
private function compose_basket( $products, $multi_item_rate ) {
    // Roll basket size per SEEDER_SPEC §2d with $multi_item_rate adjusting the tail
    $size = $this->pick_basket_size( $multi_item_rate );
    if ( $size === 1 ) {
        return [ [ 'product_id' => $this->pick_product_for_aov( $products ), 'qty' => 1 ] ];
    }
    // For multi-item baskets, sometimes force a known pair to guarantee market-basket signal
    if ( mt_rand( 0, 99 ) < 60 && $size >= 2 ) {
        $pair = self::BASKET_PAIRS[ array_rand( self::BASKET_PAIRS ) ];
        $a = $this->find_product_by_name_substring( $products, $pair[0] );
        $b = $this->find_product_by_name_substring( $products, $pair[1] );
        if ( $a && $b ) {
            return [
                [ 'product_id' => $a, 'qty' => 1 ],
                [ 'product_id' => $b, 'qty' => 1 ],
            ];
        }
    }
    // Random selection for remaining items
    $items = [];
    for ( $i = 0; $i < $size; $i++ ) {
        $items[] = [ 'product_id' => $products[ array_rand( $products ) ], 'qty' => 1 ];
    }
    return $items;
}
```
