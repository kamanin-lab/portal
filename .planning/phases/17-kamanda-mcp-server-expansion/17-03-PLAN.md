function toBlock<T>(r: PromiseSettledResult<T>): Block<T> {
  if (r.status === "fulfilled") return { status: "ok", data: r.value };
  return { status: "error", message: errorMessage(r.reason) };
}

function summarizeBriefing(p: BriefingPayload): string {
  const lines: string[] = [];

  // Run-rate (Tages-Hochrechnung)
  const runRate = p.blocks.run_rate;
  if (runRate.status === "ok") {
    const projected = runRate.data.projected_revenue.toFixed(2);
    const pace = runRate.data.pace_vs_7day_avg_pct;
    const paceStr = pace >= 0 ? `+${pace.toFixed(1)}%` : `${pace.toFixed(1)}%`;
    lines.push(
      `Tages-Hochrechnung: ${projected} (Tempo vs. 7-Tage-Schnitt ${paceStr}, Vertrauen ${runRate.data.confidence})`,
    );
  } else {
    lines.push(`Tages-Hochrechnung: n/v (Fehler: ${runRate.message})`);
  }

  // Heatmap (Bester Slot)
  const heatmap = p.blocks.heatmap;
  if (heatmap.status === "ok") {
    const dow = heatmap.data.best_slot.day_of_week;
    const hour = heatmap.data.best_slot.hour_of_day;
    const count = heatmap.data.best_slot.order_count;
    lines.push(`Bester Slot: Tag ${dow}, ${hour}:00 (${count} Bestellungen)`);
  } else {
    lines.push(`Bester Slot: n/v (Fehler: ${heatmap.message})`);
  }

  // Repeat (Stammkunden)
  const repeat = p.blocks.repeat;
  if (repeat.status === "ok") {
    const rate = repeat.data.repeat_rate_pct.toFixed(1);
    const bench = repeat.data.benchmark_pct.toFixed(1);
    lines.push(
      `Stammkunden-Quote: ${rate}% (Benchmark ${bench}%, ${repeat.data.returning_customers} von ${repeat.data.unique_customers})`,
    );
  } else {
    lines.push(`Stammkunden-Quote: n/v (Fehler: ${repeat.message})`);
  }

  // Basket (Warenkorb)
  const basket = p.blocks.basket;
  if (basket.status === "ok") {
    const mode = basket.data.mode;
    const aov = basket.data.avg_order_value.toFixed(2);
    const modeLabel =
      mode === "market_basket_product"
        ? "Warenkorb-Paare"
        : mode === "market_basket_category"
          ? "Warenkorb-Kategorien"
          : "AOV-Bänder";
    lines.push(`Warenkorb (${modeLabel}): Ø Bestellwert ${aov}`);
  } else {
    lines.push(`Warenkorb: n/v (Fehler: ${basket.message})`);
  }

  // Attention (Zahlungsaufmerksamkeit)
  const attention = p.attention;
  if (attention.status === "ok") {
    lines.push(
      `Zahlungsaufmerksamkeit: ${attention.data.total_count} Bestellungen (${attention.data.total_value.toFixed(2)} ${attention.data.currency})`,
    );
  } else {
    lines.push(`Zahlungsaufmerksamkeit: n/v (Fehler: ${attention.message})`);
  }

  return lines.join("\n");
}

function failure(
  tool: string,
  err: unknown,
): { content: [{ type: "text"; text: string }]; structuredContent: { error: string }; isError: true } {
  const message = errorMessage(err);
  return {
    content: [{ type: "text", text: `${tool} failed: ${message}` }],
    structuredContent: { error: message },
    isError: true,
  };
}

function errorMessage(err: unknown): string {
  if (err instanceof WooApiError) {
    return `WooCommerce ${err.status ?? "network"} — ${err.message}`;
  }
  if (err instanceof KmnBridgeError) {
    return `KMN bridge ${err.code} — ${err.message}`;
  }
  return err instanceof Error ? err.message : String(err);
}
