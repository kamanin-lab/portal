## JSON-RPC Envelope Contract (what the bridge client sends + receives)

Request:
```json
{
  "jsonrpc": "2.0",
  "id": "<uuid>",
  "method": "tools/call",
  "params": {
    "name": "kmn-weekly-heatmap",   // hyphenated ability name
    "arguments": { "weeks": 8 }
  }
}
```

Response (from WP MCP Adapter):
```json
{
  "jsonrpc": "2.0",
  "id": "<uuid>",
  "result": {
    "content": [{ "type": "text", "text": "<JSON-stringified envelope>" }]
  }
}
```

Envelope inside `content[0].text` (JSON.parse that string):
```json
{
  "success": true,
  "data": { /* ability-specific payload */ },
  "error": null,
  "_meta": { "plugin": "kmn-revenue-abilities", "version": "0.5.0", "calculated_at": "..." }
}
```

On failure: `{ "success": false, "data": null, "error": { "code": "...", "message": "..." }, "_meta": {...} }`.

```typescript
import { randomUUID } from "node:crypto";
import { z } from "zod";
import {
  RunRateSchema, HeatmapSchema, RepeatSchema, BasketSchema, WeeklyBriefingSchema,
  type RunRateData, type HeatmapData, type RepeatData, type BasketData, type WeeklyBriefingData,
} from "./kmn-bridge-schemas.js";

// ─── Types + Error ────────────────────────────────────────────────
export type KmnBridgeCredentials = {
  bridgeUrl: string;   // e.g. "https://summerfield.ddev.site/wp-json/mcp/kmn-revenue"
  wpUser: string;
  appPass: string;
};

export type KmnBridgeErrorCode = "timeout" | "http" | "envelope" | "schema";

export class KmnBridgeError extends Error {
  constructor(
    message: string,
    readonly code: KmnBridgeErrorCode,
    readonly endpoint: string,
    readonly status?: number,
  ) {
    super(message);
    this.name = "KmnBridgeError";
  }
}

const BRIDGE_TIMEOUT_MS = 5_000;   // per D-15

// ─── Env loader ────────────────────────────────────────────────────
export function loadKmnBridgeCredentialsFromEnv(): KmnBridgeCredentials {
  const bridgeUrl = process.env.KMN_BRIDGE_URL;
  const wpUser = process.env.WOOCOMMERCE_WP_USER;
  const appPass = process.env.WOOCOMMERCE_WP_APP_PASS;
  if (!bridgeUrl || !wpUser || !appPass) {
    throw new Error(
      "KMN bridge credentials missing. Set KMN_BRIDGE_URL, WOOCOMMERCE_WP_USER, WOOCOMMERCE_WP_APP_PASS.",
    );
  }
  return { bridgeUrl: bridgeUrl.replace(/\/+$/, ""), wpUser, appPass };
}

// ─── Redaction helper ──────────────────────────────────────────────
function safeEndpoint(creds: KmnBridgeCredentials): string {
  // Bridge URL does not carry secrets in the URL (Basic Auth is header-based),
  // so redaction is a no-op but we keep the function for parity with woocommerce.ts
  return creds.bridgeUrl;
}

// ─── Factory ───────────────────────────────────────────────────────
export function createKmnBridgeClient(
  creds: KmnBridgeCredentials = loadKmnBridgeCredentialsFromEnv(),
) {
  const endpoint = safeEndpoint(creds);
  const authHeader = "Basic " + Buffer.from(`${creds.wpUser}:${creds.appPass}`).toString("base64");

  async function callAbility<T>(
    abilityName: string,
    args: Record<string, unknown>,
    schema: z.ZodType<T>,
  ): Promise<T> {
    const rpcEnvelope = {
      jsonrpc: "2.0" as const,
      id: randomUUID(),
      method: "tools/call" as const,
      params: { name: abilityName, arguments: args },
    };

    const ctrl = new AbortController();
    const timer = setTimeout(() => ctrl.abort(), BRIDGE_TIMEOUT_MS);
    const started = Date.now();

    try {
      const res = await fetch(creds.bridgeUrl, {
        method: "POST",
        headers: {
          "content-type": "application/json",
          "accept": "application/json,text/event-stream",
          "authorization": authHeader,
        },
        body: JSON.stringify(rpcEnvelope),
        signal: ctrl.signal,
      });

      const elapsed = Date.now() - started;
      console.log(`[kmn-bridge] ${res.status} ${abilityName} ${elapsed}ms`);

      if (!res.ok) {
        const body = await res.text().catch(() => "");
        throw new KmnBridgeError(
          `KMN bridge HTTP ${res.status} calling ${abilityName}: ${body.slice(0, 200)}`,
          "http",
          endpoint,
          res.status,
        );
      }

      // JSON-RPC response: { jsonrpc, id, result: { content: [{ type: 'text', text: '<stringified envelope>' }] } }
      const rpc = await res.json() as {
        result?: { content?: Array<{ type: string; text?: string }> };
        error?: { code: number; message: string };
      };

      if (rpc.error) {
        throw new KmnBridgeError(
          `KMN bridge JSON-RPC error ${rpc.error.code} calling ${abilityName}: ${rpc.error.message}`,
          "envelope",
          endpoint,
        );
      }

      const text = rpc.result?.content?.[0]?.text;
      if (typeof text !== "string") {
        throw new KmnBridgeError(
          `KMN bridge malformed response (missing result.content[0].text) for ${abilityName}`,
          "envelope",
          endpoint,
        );
      }

      let envelope: { success?: boolean; data?: unknown; error?: { code?: string; message?: string } | null };
      try {
        envelope = JSON.parse(text);
      } catch {
        throw new KmnBridgeError(
          `KMN bridge content is not valid JSON for ${abilityName}`,
          "envelope",
          endpoint,
        );
      }

      if (envelope.success !== true) {
        const errMsg = envelope.error?.message ?? "unknown";
        throw new KmnBridgeError(
          `KMN bridge envelope error for ${abilityName}: ${errMsg}`,
          "envelope",
          endpoint,
        );
      }

      const parsed = schema.safeParse(envelope.data);
      if (!parsed.success) {
        throw new KmnBridgeError(
          `KMN bridge schema violation for ${abilityName}: ${parsed.error.message}`,
          "schema",
          endpoint,
        );
      }

      return parsed.data;
    } catch (err) {
      if (err instanceof KmnBridgeError) throw err;
      if (err instanceof Error && err.name === "AbortError") {
        throw new KmnBridgeError(
          `KMN bridge request timed out after ${BRIDGE_TIMEOUT_MS}ms at ${abilityName}`,
          "timeout",
          endpoint,
        );
      }
      const msg = err instanceof Error ? err.message : String(err);
      throw new KmnBridgeError(
        `KMN bridge request failed at ${abilityName}: ${msg}`,
        "http",
        endpoint,
      );
    } finally {
      clearTimeout(timer);
    }
  }

  return {
    runRate(): Promise<RunRateData> {
      return callAbility("kmn-revenue-run-rate", {}, RunRateSchema);
    },
    heatmap(args: { weeks: number }): Promise<HeatmapData> {
      return callAbility("kmn-weekly-heatmap", { weeks: args.weeks }, HeatmapSchema);
    },
    repeat(args: { days: number }): Promise<RepeatData> {
      return callAbility("kmn-repeat-metrics", { days: args.days }, RepeatSchema);
    },
    marketBasket(): Promise<BasketData> {
      return callAbility("kmn-market-basket", {}, BasketSchema);
    },
    weeklyBriefing(): Promise<WeeklyBriefingData> {
      return callAbility("kmn-weekly-briefing-data", {}, WeeklyBriefingSchema);
    },
  };
}

export type KmnBridgeClient = ReturnType<typeof createKmnBridgeClient>;

// Re-export inferred payload types so tool builders import from one place
export type { RunRateData, HeatmapData, RepeatData, BasketData, WeeklyBriefingData };
```
