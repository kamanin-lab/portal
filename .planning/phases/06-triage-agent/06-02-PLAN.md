async function fetchWithTimeout(
  url: string,
  options: RequestInit = {},
  timeoutMs = 10000
): Promise<Response> {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);
  try {
    return await fetch(url, { ...options, signal: controller.signal });
  } finally {
    clearTimeout(timeoutId);
  }
}
```

   const parseTriageResponse = async (resp: Response): Promise<{ parsed: TriageOutput | null; inputTokens: number; outputTokens: number }> => {
     const data = await resp.json();
     inputTokens = data.usage?.prompt_tokens ?? 0;
     outputTokens = data.usage?.completion_tokens ?? 0;
     let text = data.choices?.[0]?.message?.content || "";
     text = text.replace(/^```(?:json)?\s*/i, "").replace(/\s*```\s*$/, "").trim();
     try {
       return { parsed: JSON.parse(text), inputTokens, outputTokens };
     } catch {
       return { parsed: null, inputTokens, outputTokens };
     }
   };

   const first = await parseTriageResponse(response);
   parsed = first.parsed;
   inputTokens = first.inputTokens;
   outputTokens = first.outputTokens;

   if (!parsed) {
     // Retry once with explicit JSON instruction
     log.warn("First Claude response was not valid JSON — retrying");
     const retryResp = await fetchWithTimeout("https://openrouter.ai/api/v1/chat/completions", {
       method: "POST",
       headers: {
         "Content-Type": "application/json",
         "Authorization": `Bearer ${openrouterKey}`,
         "HTTP-Referer": "https://portal.kamanin.at",

           { role: "user", content: userMessage },
           { role: "assistant", content: "Return ONLY valid JSON, no text outside the JSON object." },
         ],
       }),
     }, 30000);
     const second = await parseTriageResponse(retryResp);
     parsed = second.parsed;
     inputTokens += second.inputTokens;
     outputTokens += second.outputTokens;
   }
   ```
